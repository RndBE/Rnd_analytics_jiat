    <script>
        // Parameter dasar fisik visualisasi disesuaikan untuk skala Dummy API (TMA ~3.5m - 5.8m)
        const KEDALAMAN_SUMUR = 10.0;

        // Elements
        const waterElement = document.getElementById('dynamic-well-water');
        const tmaLabelElement = document.getElementById('label-tma-dynamic');
        const tmaContainer = document.getElementById('tma-current');
        
        function updateWellGraphic(tmaValue) {
            // Sisa Kedalaman Air Berbanding Dasar Sumur 
            const ketinggian_air_m = KEDALAMAN_SUMUR - tmaValue;
            
            // Translate meter to percentage 0-100% of the pipe container
            const heightPercentage = Math.max(0, (ketinggian_air_m / KEDALAMAN_SUMUR) * 100);
            
            // Translate margin top TMA
            const topMarginPercentage = Math.min(90, (tmaValue / KEDALAMAN_SUMUR) * 100);

            waterElement.style.height = `${heightPercentage}%`;
            tmaLabelElement.style.top = `calc(${topMarginPercentage}% + 2rem)`;
            tmaLabelElement.innerHTML = `<span>TMA Aktual (${tmaValue.toFixed(2)}m)</span>`;
            
            // Update UI Data Card Sumur serentak
            tmaContainer.innerText = tmaValue.toFixed(3);
            if (tmaValue > 4.5) { 
                tmaContainer.style.color = '#ef4444'; 
            } else {
                tmaContainer.style.color = '#10b981'; 
            }
        }

        let chartInstance = null;

        async function initDashboard() {
            try {
                // 1. Menarik data simulasi 24 jam dari API Dummy Python (Port 8001)
                const res = await fetch("http://localhost:8001/simulasi/siklus?resolusi_menit=60");
                const payload = await res.json();
                const telemetryData = payload.data;

                const labels = [];
                const dataPumping = [];
                const dataThreshold = [];
                const limitYield = payload.max_drawdown;

                telemetryData.forEach(point => {
                    labels.push(point.timestamp);
                    dataPumping.push(point.tma_meter);
                    dataThreshold.push(limitYield);
                });

                // ============================================
                // LOOP SIMULASI ANIMASI TABUNG & PERGERAKAN AI
                // ============================================
                let currentIter = 0;
                updateWellGraphic(telemetryData[0].tma_meter);
                
                setInterval(() => {
                    currentIter++;
                    if(currentIter >= telemetryData.length) {
                        currentIter = 0; 
                    }
                    let cursorData = telemetryData[currentIter];
                    updateWellGraphic(cursorData.tma_meter);
                    
                    // Update Panel AI berdasarkan API Dummy respon per jam itu
                    const recTimeEl = document.getElementById('ai-recovery-time');
                    const transEl = document.getElementById('ai-transmissivity');
                    
                    // Ganti Header Box kedua dari Transmisivitas jadi Persentase Pulih
                    transEl.parentElement.previousElementSibling.innerText = "Progres Pemulihan";
                    
                    if (cursorData.fase === 'RECOVERY') {
                         recTimeEl.innerText = cursorData.eta_formatted;
                         recTimeEl.style.fontSize = "1.4rem";
                         recTimeEl.style.color = "#facc15"; // Kuning
                         transEl.innerText = `${cursorData.persen_recovery}%`;
                    } else if (cursorData.fase === 'DISCHARGING') {
                         recTimeEl.innerText = "Pompa Aktif Surut";
                         recTimeEl.style.fontSize = "1.6rem";
                         recTimeEl.style.color = "#ef4444"; // Merah
                         transEl.innerText = "0%";
                    } else {
                         recTimeEl.innerText = "Sudah Stabil";
                         recTimeEl.style.color = "#10b981"; // Hijau
                         transEl.innerText = "100%";
                    }
                    recTimeEl.nextElementSibling.style.display = 'none';

                }, 1000); // 1 detik web = 1 jam di simulasi dunia nyata

                // ============================================
                // SETUP GRAFIK MENGGUNAKAN CHART.JS
                // ============================================
                const ctx = document.getElementById('tmaChart').getContext('2d');
                const gradientFill = ctx.createLinearGradient(0, 0, 0, 450);
                gradientFill.addColorStop(0, 'rgba(59, 130, 246, 0.5)'); 
                gradientFill.addColorStop(1, 'rgba(59, 130, 246, 0.05)'); 

                chartInstance = new Chart(ctx, {
                    type: 'line',
                    data: {
                        labels: labels,
